// Nova DevDocs — Core Integration
const bedrock = new BedrockRuntimeClient({ region: 'us-east-1' });

// Step 1: Nova 2 Sonic captures and transcribes voice
const sonicRes = await bedrock.send(new InvokeModelWithResponseStreamCommand({
  modelId: 'amazon.nova-sonic-v1:0',
  body: JSON.stringify({ inputAudio: audioStream, systemPrompt })
}));

// Step 2: Nova 2 Lite reasons over your knowledge base
const liteRes = await bedrock.send(new InvokeModelCommand({
  modelId: 'amazon.nova-lite-v1:0',
  body: JSON.stringify({
    messages: conversationHistory,
    tools: [searchDocsTool, fetchFileTool],
    system: systemPrompt + kbContext
  })
}));

// Step 3: Synthesized answer spoken back via Nova 2 Sonic
await streamVoiceResponse(liteRes.answer, sonicStream);
